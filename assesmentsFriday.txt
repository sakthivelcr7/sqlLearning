9a. Select customer first_name/last_name and actor first_name/last_name columns from 
performing a /left join/ 
# between the customer and actor column on the last_name column in each table. 
# (i.e. `customer.last_name = actor.last_name`)
# Label customer first_name/last_name columns as customer_first_name/customer_last_name
# Label actor first_name/last_name columns in a similar fashion.

ans: 
select customer.first_name as customer_first_name ,customer.last_name as customer_last_name,
a.first_name,a.last_name from 
customer  left join actor a on customer.last_name= a.last_name;

# 9b. Select the customer first_name/last_name and actor first_name/last_name 
columns from performing a /right join between the customer and actor column
 on the last_name column in each table. (i.e. `customer.last_name = actor.last_name`)
ans:
select customer.first_name ,customer.last_name ,
a.first_name,a.last_name from 
customer  right join actor a on customer.last_name= a.last_name;


# 9c. Select the customer first_name/last_name and actor first_name/last_name 
columns from performing an inner join between the customer and actor column
 on the last_name column in each table. (i.e. `customer.last_name = actor.last_name`)
ans:
select customer.first_name ,customer.last_name ,
a.first_name,a.last_name from 
customer  inner join actor a on customer.last_name= a.last_name;

9d. Select the city name and country name columns from 
the city table, performing a left join with the country 
table to get the country name column.
ans:
select city.city ,country.country from
city  left join country on city.country_id= country.country_id;

# 9e. Select the title, description, release year, and language name
 columns from the film table, performing a left join with the language
 table to get the "language" column.
# Label the language.name column as "language" (e.g. `select language.name as language`)
ans:
select film.title ,film. description,film.release_year ,language.name from
film left join language on film.language_id=language.language_id;


# 9f. Select the first_name, last_name, address, address2, 
city name, district, and postal code columns from the staff table,
 performing 2 left joins with the address table then the city table to
 get the address and city related columns.
ans:
select staff.first_name,staff.last_name,address.address,
address.address2,address.district,address.postal_code from address 
left join staff on address.address_id=staff.address_id left
 join  city  on city.city_id=address.city_id;
